\begin{multicols}{3}
\begin{itemize}
    \item Status
    \item From
    \item To
    \item Type
    \item Value
    \item Gas
    \item Gas price
    \item Max fee per gas
    \item Max priority fee per gas
\end{itemize}
\end{multicols}
